function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var parsed = JSON.parse(e.postData.contents); var sheet; var row; if (parsed.tipo == "cliente") { sheet = ss.getSheetByName("Clientes"); row = [new Date(), parsed.nombre, parsed.empresa, parsed.rut, parsed.correo, parsed.telefono, parsed.direccion]; sheet.appendRow(row); } if (parsed.tipo == "cotizacion") { sheet = ss.getSheetByName("Cotizaciones"); row = [new Date(), parsed.nombre, parsed.correo, parsed.telefono, parsed.producto, parsed.cantidad, parsed.unidad, parsed.modalidad, parsed.precioUnitario, parsed.neto, parsed.iva, parsed.total]; sheet.appendRow(row); enviarCorreo("Cotización", parsed); } if (parsed.tipo == "consulta") { sheet = ss.getSheetByName("Consultas"); row = [new Date(), parsed.nombre, parsed.empresa, parsed.correo, parsed.telefono, parsed.mensaje]; sheet.appendRow(row); enviarCorreo("Consulta", parsed); } return ContentService.createTextOutput(JSON.stringify({resultado: "ok"})).setMimeType(ContentService.MimeType.JSON); } function enviarCorreo(tipo, datos) { var destino = "comercial@we-trade.cl"; var asunto = "Nueva " + tipo + " - La Carbonería"; var cuerpo = "Has recibido una nueva " + tipo + " desde la página web:\n\n"; for (var key in datos) { if (key != "tipo") { cuerpo += key + ": " + datos[key] + "\n"; } } MailApp.sendEmail(destino, asunto, cuerpo); }
